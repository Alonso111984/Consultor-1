                                            />
                                        </div>

                                        <div>
                                            <label className="block text-[11px] font-bold text-slate-500 uppercase mb-1">Calle Secundaria / Referencia</label>
                                            <input 
                                                required 
                                                type="text" 
                                                placeholder="Ej: Junto a la farmacia o calle secundaria" 
                                                value={formData.secondaryStreet}
                                                className="w-full p-3 text-sm bg-slate-50 rounded-xl border border-slate-200 focus:outline-none focus:ring-2 focus:ring-indigo-500/20 focus:border-indigo-500 transition-all" 
                                                onChange={e => handleInputChange('secondaryStreet', e.target.value)} 
                                            />
                                        </div>
                                    </div>

                                    {/* Botón de Enlace Directo a WhatsApp */}
                                    <button 
                                        type="submit" 
                                        className="w-full bg-slate-900 text-white p-4 rounded-xl font-bold text-sm shadow-xl hover:bg-slate-800 transition-all active:scale-95 mt-4 flex items-center justify-center gap-2"
                                    >
                                        <svg className="w-5 h-5 fill-current text-white" viewBox="0 0 24 24">
                                            <path d="M.057 24l1.687-6.163c-1.041-1.804-1.588-3.849-1.587-5.946C.06 5.348 5.397.01 12.008.01c3.202.001 6.212 1.246 8.477 3.514 2.266 2.268 3.507 5.28 3.505 8.484-.004 6.657-5.34 11.997-11.953 11.997-2.005-.001-3.973-.503-5.727-1.458L0 24zm6.59-4.846c1.6.95 3.188 1.449 4.825 1.451 5.436 0 9.86-4.37 9.864-9.799.002-2.63-1.023-5.101-2.885-6.968C16.59 2.016 14.11 1.008 11.48 1.008c-5.44 0-9.866 4.372-9.87 9.802 0 1.714.475 3.387 1.378 4.842l-.994 3.63 3.733-.968c1.529.875 3.063 1.31 4.92 1.31z"></path>
                                        </svg>
                                        QUIERO MI PEDIDO
                                    </button>
                                </form>
                            </div>
                        )}

                        {step === 'chat' && (
                            <div className="flex flex-col h-[500px] bg-slate-50 rounded-3xl border border-slate-200 overflow-hidden animate-fade-in shadow-inner">
                                {/* Chat Header */}
                                <div className="bg-white border-b border-slate-200 p-4 flex items-center justify-between">
                                    <div className="flex items-center gap-3">
                                        <div className="bg-indigo-100 p-2 rounded-full text-indigo-600">
                                            {IconMap.bot()}
                                        </div>
                                        <div>
                                            <h3 className="font-bold text-sm text-slate-800">Experto IA</h3>
                                            <p className="text-[10px] text-emerald-600 font-semibold flex items-center gap-1">
                                                <span className="w-1.5 h-1.5 bg-emerald-500 rounded-full animate-pulse"></span>
                                                En línea
                                            </p>
                                        </div>
                                    </div>
                                    <button 
                                        onClick={() => setStep('home')}
                                        className="text-slate-400 hover:text-slate-600 text-xs font-semibold"
                                    >
                                        Volver
                                    </button>
                                </div>

                                {/* Chat Messages Area */}
                                <div className="flex-1 overflow-y-auto p-4 space-y-4 scroll-smooth">
                                    {chatHistory.map((msg, idx) => (
                                        <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
                                            <div className={`max-w-[85%] p-3 rounded-2xl text-sm ${
                                                msg.role === 'user' 
                                                    ? 'bg-indigo-600 text-white rounded-tr-sm' 
                                                    : 'bg-white border border-slate-200 text-slate-700 rounded-tl-sm shadow-sm'
                                            }`}>
                                                {msg.text}
                                            </div>
                                        </div>
                                    ))}
                                    
                                    {isChatLoading && (
                                        <div className="flex justify-start">
                                            <div className="bg-white border border-slate-200 p-4 rounded-2xl rounded-tl-sm shadow-sm flex items-center gap-1">
                                                <svg height="8" width="24" viewBox="0 0 24 8">
                                                    <circle className="typing-dot" cx="4" cy="4" r="3" />
                                                    <circle className="typing-dot" cx="12" cy="4" r="3" />
                                                    <circle className="typing-dot" cx="20" cy="4" r="3" />
                                                </svg>
                                            </div>
                                        </div>
                                    )}
                                    <div ref={chatEndRef} />
                                </div>

                                {/* Chat Input Area */}
                                <div className="p-3 bg-white border-t border-slate-200">
                                    <form onSubmit={handleSendMessage} className="flex gap-2">
                                        <input 
                                            type="text" 
                                            value={chatInput}
                                            onChange={(e) => setChatInput(e.target.value)}
                                            placeholder="Describe lo que buscas..."
                                            className="flex-1 bg-slate-50 border border-slate-200 rounded-xl px-4 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-indigo-500/30 transition-all"
                                            disabled={isChatLoading}
                                        />
                                        <button 
                                            type="submit"
                                            disabled={!chatInput.trim() || isChatLoading}
                                            className="bg-indigo-600 text-white p-2.5 rounded-xl hover:bg-indigo-700 disabled:opacity-50 disabled:active:scale-100 active:scale-95 transition-all flex-shrink-0"
                                        >
                                            {IconMap.send()}
                                        </button>
                                    </form>
                                    <div className="text-center mt-2">
                                         <button 
                                            onClick={() => {
                                                setRecommended(PRODUCTS[0]); // Default to first product if they just want to buy from chat
                                                setStep('cart');
                                            }}
                                            className="text-[10px] font-bold text-indigo-600 uppercase tracking-wider hover:underline"
                                        >
                                            Ya sé lo que quiero → Ir al pedido
                                        </button>
                                    </div>
                                </div>
                            </div>
                        )}

                    </main>

                    {/* Footer de Confianza */}
                    <footer className="p-4 bg-slate-100 text-center flex flex-col gap-1.5 border-t border-slate-200">
                        <span className="text-[10px] text-slate-600 font-bold tracking-wider">🛡️ COMPRA 100% SEGURA CONTRA ENTREGA EN ECUADOR</span>
                        <span className="text-[9px] text-slate-400 font-semibold tracking-wider">🚚 ENVÍOS A TODO EL PAÍS EN 3 DÍAS LABORABLES</span>
                    </footer>
                </div>
            );
        }
